{
        public frmOrders()
        {
            InitializeComponent();
        }

        private void btnOrdersTonen_Click(object sender, EventArgs e)
        {
            //Openen van de connectie
            scnnArtemisSQL.Open();

            //Aantal orders weergeven in de label
            scmdOrdersTonen.CommandText = "Select count(*) from tblOrders";
            lblOrders.Text = scmdOrdersTonen.ExecuteScalar().ToString();

            //Connectie sluiten
            scnnArtemisSQL.Close(); 
        }

        private void btnOrdersAanpassen_Click(object sender, EventArgs e)
        {

            //Declaratie van de variabelen
            int intAantal;
            
            //Openen van de connectie
            scnnArtemisSQL.Open();

            //het verdubbelen van de vrachtprijs uitvoeren
            intAantal = scmdOrdersVermigvuldigen.ExecuteNonQuery(); 

            //Connectie sluiten
            scnnArtemisSQL.Close();

            //MessageBox laten verschijnen
            MessageBox.Show(intAantal.ToString() + " records aangepast"); 


        }

        private void btnLadenInLijst_Click(object sender, EventArgs e)
        {
            //Openen connectie
            scnnArtemisSQL.Open();
            SqlDataReader sdrOrders = scmdOrdersInlijst.ExecuteReader();

            //Orders tonen in lijst
            while(sdrOrders.Read())
            {
                lbOrders.Items.Add(sdrOrders["OrderID"].ToString());
            }

            //Sluiten connectie
            sdrOrders.Close();
            scnnArtemisSQL.Close(); 
        }
    }
}
